function draw() { const canvas = document.getElementById("canvas"); if (!canvas.getContext) return; const ctx = canvas.getContext("2d"); /* ====================== RECTÁNGULOS ====================== */ ctx.fillRect(25, 25, 100, 100); ctx.clearRect(45, 45, 60, 60); ctx.strokeRect(50, 50, 50, 50); /* ====================== TRIÁNGULO ====================== */ ctx.beginPath(); ctx.moveTo(200, 25); ctx.lineTo(250, 125); ctx.lineTo(150, 125); ctx.fill(); /* ====================== HAPPY FACE ====================== */ ctx.beginPath(); ctx.arc(400, 75, 50, 0, Math.PI * 2); ctx.moveTo(435, 75); ctx.arc(400, 75, 35, 0, Math.PI); ctx.moveTo(385, 65); ctx.arc(380, 65, 5, 0, Math.PI * 2); ctx.moveTo(425, 65); ctx.arc(420, 65, 5, 0, Math.PI * 2); ctx.stroke(); /* ====================== LÍNEAS ====================== */ ctx.beginPath(); ctx.moveTo(25, 200); ctx.lineTo(125, 200); ctx.lineTo(75, 260); ctx.stroke(); /* ====================== ARCOS ====================== */ ctx.beginPath(); ctx.arc(220, 230, 40, 0, Math.PI * 1.5); ctx.stroke(); /* ====================== CURVA CUADRÁTICA ====================== */ ctx.beginPath(); ctx.moveTo(350, 200); ctx.quadraticCurveTo(400, 300, 450, 200); ctx.stroke(); /* ====================== CURVA BÉZIER ====================== */ ctx.beginPath(); ctx.moveTo(500, 200); ctx.bezierCurveTo(520, 300, 580, 100, 650, 200); ctx.stroke(); /* ====================== COMBINACIÓN FIGURAS ====================== */ ctx.beginPath(); ctx.rect(720, 180, 100, 100); ctx.arc(770, 230, 40, 0, Math.PI * 2); ctx.stroke(); }
